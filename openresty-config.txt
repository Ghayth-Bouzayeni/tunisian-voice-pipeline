apiVersion: v1
kind: ConfigMap
metadata:
  name: openresty-config
data:
  nginx.conf: |
    worker_processes 1;

    events {
        worker_connections 1024;
    }

    http {
        upstream agg_api_backend {
            server agg-api-service:9000;
        }

        server {
            listen 8080;

            # Redirection de toutes les requêtes vers le backend avec timeout prolongé
            location / {
                proxy_pass http://agg_api_backend;
                proxy_set_header Host $host;
                proxy_set_header X-Real-IP $remote_addr;
                proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
                proxy_set_header X-Forwarded-Proto $scheme;

                # Timeouts pour éviter le 504 Gateway Time-out
                proxy_connect_timeout 3600s;
                proxy_send_timeout 3600s;
                proxy_read_timeout 3600s;
                send_timeout 3600s;

                # Optionnel : augmenter la taille du buffer pour les grosses réponses
                proxy_buffer_size 16k;
                proxy_buffers 4 32k;
                proxy_busy_buffers_size 64k;
            }
        }
    }
